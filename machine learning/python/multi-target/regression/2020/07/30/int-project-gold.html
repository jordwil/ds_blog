def calc_recovery(C, F, T, conv_inf=True, conv_neg=True, conv_high=True):
    rec = (( C*(F - T) ) / ( F*(C - T) )) * 100
    if conv_inf:
        rec[np.abs(rec) == np.inf] = np.nan
    if conv_neg:
        rec[rec < 0] = np.nan
        
    if conv_high:
        rec[rec > 100] = np.nan
    return rec

def smape_score(y, y_pred):
    """Symmetric Mean Absolute Percentage Error"""
    smape_r = np.mean( (np.abs(y_pred[:,0] - y[:,0]) / ((np.abs(y[:,0]) + np.abs(y_pred[:,0]))/2))) *100
    smape_f = np.mean( (np.abs(y_pred[:,1] - y[:,1]) / ((np.abs(y[:,1]) + np.abs(y_pred[:,1]))/2))) *100
    final_score = smape_r*.25 + smape_f*.75
    return final_score
